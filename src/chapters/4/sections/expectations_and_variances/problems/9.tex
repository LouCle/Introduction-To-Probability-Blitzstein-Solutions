\frac{1}{4}64000 = 20500. \text{Var}{X} = \text{E}(X^{2}) - (\text{E}(X))^{2}

\item $\text{E}(X) = \frac{3}{4}1000 + \frac{1}{4}\frac{1}{2}32000 + \frac{1}

\end{enumerate}
